    QPUの古典レジスタのクラス

    Attributes:
        count (int): 内包するレジスタの数
        values (List[float]): レジスタの値リスト
    """

    def __init__(self, count: int):
        """
        Args:
            count (int): 初期化するレジスタの数
        """
        values = [None for index in range(count)]  # type: List[Optional[float]]
        self.count = count
        self.values = values
